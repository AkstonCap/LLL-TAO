237  if(memory::compare((uint8_t *)HEADER, (uint8_t *)(config::fTestNet ? MESSAGE_START_TESTNET : MESSAGE_START_MAINNET), sizeof(HEADER)) != 0)
238  return debug::error("Message Packet (Invalid Packet Header");
239 
240  /* Make sure Packet length is within bounds. (Max 512 MB Packet Size) */
241  if (LENGTH > (1024 * 1024 * 512))
242  return debug::error("Message Packet (", MESSAGE, ", ", LENGTH, " bytes) : Message too Large");
243 
244  /* Double check the Message Checksum. */
245  uint512_t hash = LLC::SK512(DATA.begin(), DATA.end());
246  uint32_t nChecksum = 0;
247  //memcpy(&nChecksum, &hash, sizeof(nChecksum));

249
