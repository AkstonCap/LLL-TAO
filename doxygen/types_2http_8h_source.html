130  void ReadPacket() final
131  {
132  if(!INCOMING.Complete())
133  {
134  /* Handle Reading Data into Buffer. */
135  uint32_t nAvailable = Available();
136  if(nAvailable > 0)
137  {
138  std::vector<int8_t> vchData(nAvailable);
139  uint32_t nRead = Read(vchData, nAvailable);
140  if(nRead > 0)
141  vchBuffer.insert(vchBuffer.end(), vchData.begin(), vchData.begin() + nRead);
142  }
143 
144  /* If waiting for buffer data, don't try to parse. */
145  if(vchBuffer.size() == 0)
146  return;
147 
148  /* Allow up to 10 iterations to parse the header. */
149  for(int i = 0; i < 10; i++)
150  {
151  /* Read content if there is some. */
152  if(INCOMING.fHeader)
153  {
154  INCOMING.strContent += std::string(vchBuffer.begin(), vchBuffer.end());
155  vchBuffer.clear();
156  return;
157  }
158 
159  /* Break out the lines by the input buffer. */
160  auto it = std::find(vchBuffer.begin(), vchBuffer.end(), '\n');
161 
162  /* Return if a full line hasn't been read yet. */
163  if(it == vchBuffer.end())
164  return;
165 
166  /* Check for the end of header with double CLRF. */
167  if(vchBuffer.begin() == (it - 1))
168  {
169  INCOMING.fHeader = true;
170 
171  vchBuffer.erase(vchBuffer.begin(), it + 1); //erase the CLRF
172  //this->Event()
173  //TODO: assess the events code and calling virutal method from lower class in the inheritance heirarchy
174  }
175 
176  /* Read all the headers. */
177  else if(!INCOMING.fHeader)
178  {
179  /* Extract the line from the buffer. */
180  std::string strLine = std::string(vchBuffer.begin(), it - 1);
181 
182  /* Dump the header if requested on read. */
183  if(config::GetBoolArg("-httpheader"))
184  debug::log(0, strLine);
185 
186  /* Find the delimiter to split. */
187  std::string::size_type pos = strLine.find(':', 0);
188 
189  /* Handle the request types. */
190  if(INCOMING.strType == "")
191  {
192  /* Find the end of request type. */
193  std::string::size_type npos = strLine.find(' ', 0);
194  INCOMING.strType = strLine.substr(0, npos);
195 
196  /* Find the start of version. */
197  std::string::size_type npos2 = strLine.find(' ', npos + 1);
198  INCOMING.strVersion = strLine.substr(npos2 + 1);
199 
200  /* Parse request from between the two. */
201  INCOMING.strRequest = strLine.substr(npos + 1, npos2 - INCOMING.strType.length() - 1);
202 
203  }
204 
205  /* Handle normal headers. */
206  else if(pos != std::string::npos)
207  {
208  /* Set the field value to lowercase. */
209  std::string field = ToLower(strLine.substr(0, pos));
210 
211  /* Parse out the content length field. */
212  if(field == "content-length")
213  INCOMING.nContentLength = stoi(strLine.substr(pos + 2));
214 
215  /* Add line to the headers map. */
216  INCOMING.mapHeaders[field] = strLine.substr(pos + 2);
217 
218  }
219 
220  /* Erase line read from the read buffer. */
221  vchBuffer.erase(vchBuffer.begin(), it + 1);
222  }
223  }
224  }
225  }
