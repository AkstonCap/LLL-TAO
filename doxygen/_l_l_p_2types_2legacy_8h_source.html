209  void ReadPacket() final
210  {
211  if(!INCOMING.Complete())
212  {
214  if(INCOMING.IsNull() && Available() >= 24)
215  {
216  std::vector<uint8_t> BYTES(24, 0);
217  if(Read(BYTES, 24) == 24)
218  {
219  DataStream ssHeader(BYTES, SER_NETWORK, MIN_PROTO_VERSION);
220  ssHeader >> INCOMING;
221 
222  Event(EVENT_HEADER);
223  }
224  }
225 
227  uint32_t nAvailable = Available();
228  if(nAvailable > 0 && !INCOMING.IsNull() && INCOMING.DATA.size() < INCOMING.LENGTH)
229  {
230  /* Create the packet data object. */
231  std::vector<uint8_t> DATA( std::min( nAvailable, (uint32_t)(INCOMING.LENGTH - INCOMING.DATA.size())), 0);
232 
233  /* Read up to 512 bytes of data. */
234  if(Read(DATA, DATA.size()) == DATA.size())
235  {
236  INCOMING.DATA.insert(INCOMING.DATA.end(), DATA.begin(), DATA.end());
237  Event(EVENT_PACKET, DATA.size());
238  }
239  }
240  }

252  void PushGetBlocks(const uint1024_t& hashBlockFrom, const uint1024_t& hashBlockTo)
253  {
254  /* Filter out duplicate requests. */
255  if(hashLastGetblocks == hashBlockFrom && nLastGetBlocks + 10 > runtime::timestamp())
256  return;
257 
258  /* Update the last timestamp this was called. */
259  nLastGetBlocks = runtime::timestamp();
260 
261  /* Update the hash that was used for last request. */
262  hashLastGetblocks = hashBlockFrom;
263 
264  /* Push the request to the node. */
265  PushMessage("getblocks", Legacy::Locator(hashBlockFrom), hashBlockTo);
266 
267  /* Debug output for monitoring. */
268  debug::log(0, NODE, "requesting getblocks from ", hashBlockFrom.ToString().substr(0, 20), " to ", hashBlockTo.ToString().substr(0, 20));
269  }
