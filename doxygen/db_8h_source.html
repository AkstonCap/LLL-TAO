159  template<typename K, typename T>
160  bool Read(const K& key, T& value)
161  {
162  if (pdb == nullptr)
163  return false;
164 
165  /* Key */
166  DataStream ssKey(SER_DISK, LLD::DATABASE_VERSION);
167  ssKey.reserve(1000);
168  ssKey << key;
169  Dbt datKey(ssKey.data(), ssKey.size());
170 
171  /* Value */
172  Dbt datValue;
173  datValue.set_flags(DB_DBT_MALLOC); // Berkeley will allocate memory for returned value, will need to free before return
174 
175  /* Read */
176  int32_t ret = pdb->get(GetTxn(), &datKey, &datValue, 0);
177 
178  /* Clear the key memory */
179  memset(datKey.get_data(), 0, datKey.get_size());
180 
181  if (datValue.get_data() == nullptr)
182  return false; // Key not found or no value present, can safely return without free() because there is nothing to free
183 
184  /* Unserialize value */
185  try {
186  /* get_data returns a void* and currently uses a C-style cast to load into DataStream */
187  DataStream ssValue((char*)datValue.get_data(), (char*)datValue.get_data() + datValue.get_size(), SER_DISK, LLD::DATABASE_VERSION);
188  ssValue >> value;
189  }
190  catch (std::exception &e) {
191  /* Still need to free any memory allocated for datValue, so do not return here. Just set ret so it returns false */
192  ret = -1;
193  }
194 
195  /* Clear and free value memory */
196  memset(datValue.get_data(), 0, datValue.get_size());
197  free(datValue.get_data());
198 
199  return (ret == 0);
200  }
