351  void ReadPacket() final
352  {
353 
354  /* Handle Reading Packet Type Header. */
355  if(Available() >= 1 && INCOMING.IsNull())
356  {
357  std::vector<uint8_t> HEADER(1, 255);
358  if(Read(HEADER, 1) == 1)
359  INCOMING.HEADER = HEADER[0];
360  }
361 
362  /* Read the packet length. */
363  if(Available() >= 4 && INCOMING.LENGTH == 0)
364  {
365  /* Handle Reading Packet Length Header. */
366  std::vector<uint8_t> BYTES(4, 0);
367  if(Read(BYTES, 4) == 4)
368  {
369  INCOMING.SetLength(BYTES);
370  Event(EVENT_HEADER);
371  }
372  }
373 
374  /* Handle Reading Packet Data. */
375  uint32_t nAvailable = Available();
376  if(nAvailable > 0 && INCOMING.LENGTH > 0 && INCOMING.DATA.size() < INCOMING.LENGTH)
377  {
378  /* Read the data in the packet */
379  std::vector<uint8_t> DATA( std::min(nAvailable, (uint32_t)(INCOMING.LENGTH - INCOMING.DATA.size())), 0);
380 
381  /* On successful read, fire event and add data to packet. */
382  if(Read(DATA, DATA.size()) == DATA.size())
383  {
384  INCOMING.DATA.insert(INCOMING.DATA.end(), DATA.begin(), DATA.end());
385  Event(EVENT_PACKET, DATA.size());
386  }
387  }
388  }
