337  bool Erase(const Key& key)
338  {
339  if(nFlags & FLAGS::READONLY)
340  assert(!"Erase called on database in read-only mode");
341 
342  /* Serialize Key into Bytes. */
343  DataStream ssKey(SER_LLD, DATABASE_VERSION);
344  ssKey << key;
345 
346  /* Remove the item from the cache pool. */
347  cachePool->Remove(ssKey.Bytes());
348 
349  /* Add transaction to erase queue. */
350  { LOCK(TRANSACTION_MUTEX);
351  if(pTransaction)
352  {
353  /* Create an append only stream. */
354  std::ofstream stream(debug::safe_printstr(config::GetDataDir(), strName, "/journal.dat"), std::ios::app | std::ios::binary);
355 
356  /* Serialize the key. */
357  DataStream ssJournal(SER_LLD, DATABASE_VERSION);
358  ssJournal << std::string("erase") << ssKey.Bytes();
359 
360  /* Write to the file. */
361  const std::vector<uint8_t>& vBytes = ssJournal.Bytes();
362  stream.write((char*)&vBytes[0], vBytes.size());
363  stream.close();
364 
365  /* Erase the transaction data. */
366  pTransaction->EraseTransaction(ssKey.Bytes());
367 
368  return true;
369  }
370  }
371 
372  /* Return the Key existance in the Keychain Database. */
373  bool fErased = pSectorKeys->Erase(ssKey.Bytes());
374 
375  if(config::GetBoolArg("-runtime", false))
376  {
377  debug::log(0, ANSI_COLOR_GREEN FUNCTION, "executed in ",
378  runtime.ElapsedMicroseconds(), " micro-seconds" ANSI_COLOR_RESET);
379  }

602  bool Get(const std::vector<uint8_t>& vKey, std::vector<uint8_t>& vData)
603  {
604  /* Iterate if meters are enabled. */
605  nBytesRead += (vKey.size() + vData.size());
606 
607  /* Check the cache pool for key first. */
608  if(cachePool->Get(vKey, vData))
609  return true;
610 
611  /* Get the key from the keychain. */
612  SectorKey cKey;
613  if(pSectorKeys->Get(vKey, cKey))
614  {
615  { LOCK(SECTOR_MUTEX);
616 
617  /* Find the file stream for LRU cache. */
618  std::fstream* pstream;
619  if(!fileCache->Get(cKey.nSectorFile, pstream))
620  {
621  /* Set the new stream pointer. */
622  pstream = new std::fstream(debug::strprintf("%s_block.%05u", strBaseLocation.c_str(), cKey.nSectorFile), std::ios::in | std::ios::out | std::ios::binary);
623  if(!pstream->is_open())
624  {
625  delete pstream;
626  return debug::error(FUNCTION, "couldn't create stream file");
627  }
628 
629  /* If file not found add to LRU cache. */
630  fileCache->Put(cKey.nSectorFile, pstream);
631  }
632 
633  /* Seek to the Sector Position on Disk. */
634  pstream->seekg(cKey.nSectorStart, std::ios::beg);
635  vData.resize(cKey.nSectorSize);
636 
637  /* Read the State and Size of Sector Header. */
638  if(!pstream->read((char*) &vData[0], vData.size()))
639  debug::error(FUNCTION, "only ", pstream->gcount(), "/", vData.size(), " bytes read");
640 
641  }
642 
643  /* Add to cache */
644  cachePool->Put(vKey, vData);
645 
646  /* Verbose Debug Logging. */
647  debug::log(5, FUNCTION, "Current File: ", cKey.nSectorFile,
648  " | Current File Size: ", cKey.nSectorStart, "\n", HexStr(vData.begin(), vData.end(), true));
649 
650  return true;
651  }
652 
653  return false;

848  bool Put(const std::vector<uint8_t>& vKey, const std::vector<uint8_t>& vData)
849  {
850  /* Write the data into the memory cache. */
851  cachePool->Put(vKey, vData, !(nFlags & FLAGS::FORCE));
852 
853  /* Handle force write mode. */
854  if(nFlags & FLAGS::FORCE)
855  return Force(vKey, vData);
856 
857  /* Wait if the buffer is full. */
