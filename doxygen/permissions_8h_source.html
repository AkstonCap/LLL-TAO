26 inline bool CheckPermissions(std::string strAddress, uint32_t nPort)
27 {
28  /* Bypass localhost addresses first. */
29  if(strAddress == "127.0.0.1" || strAddress == "::1")
30  return true;
31 
32  /* Split the Address into String Vector. */
33  std::vector<std::string> vAddress = Split(strAddress, '.');
34  if(vAddress.size() != 4)
35  return debug::error("Address size not at least 4 bytes.");
36 
37  /* Check against the commandline parameters. */
38  const std::vector<std::string>& vAllow = config::mapMultiArgs["-llpallowip"];
39  for(int nIndex = 0; nIndex < vAllow.size(); ++nIndex)
40  {
41  /* Detect if the port for LLP filtering is a wildcard or not. */
42  bool fWildcardPort = (vAllow[nIndex].find(":") == std::string::npos);
43 
44  /* Scan out the data for the wildcard port. */
45  std::vector<std::string> vCheck = Split(vAllow[nIndex], ',');
46 
47  /* Skip invalid inputs. */
48  if(vCheck.size() != 4)
49  continue;
50 
51  /* Check the Wildcard port. */
52  if(!fWildcardPort)
53  {
54  std::vector<std::string> strPort = Split(vCheck[3], ':');
55  vCheck[3] = strPort[0];
56 
57  uint32_t nPortCheck = stoi(strPort[1]);
58  if(nPort != nPortCheck)
59  return debug::error("Bad Port.");
60  }
61 
62  /* Check the components of IP address. */
63  for(int nByte = 0; nByte < 4; ++nByte)
64  if(vCheck[nByte] != "*" && vCheck[nByte] != vAddress[nByte])
65  return debug::error("Check ", vCheck[nByte], " - ", vAddress[nByte]);
66  }
67

73 inline bool WildcardMatch(const char* psz, const char* mask)
74 {
75  for(;;)
76  {
77  switch (*mask)
78  {
79  case '\0':
80  return (*psz == '\0');
81  case '*':
82  return WildcardMatch(psz, mask+1) || (*psz && WildcardMatch(psz+1, mask));
83  case '?':
84  if (*psz == '\0')
85  return false;
86  break;
87  default:
88  if (*psz != *mask)
89  return false;
90  break;
91  }
92  ++psz;
93  ++mask;
94  }
95 }
