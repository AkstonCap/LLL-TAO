131  uint32_t GetBucket(const std::vector<uint8_t> &vKey) const
132  {
133  uint64_t nBucket = 0;
134  for(uint32_t i = 0; i < vKey.size() && i < 8; ++i)
135  nBucket += vKey[i] << (8 * i);
136 
137  return nBucket % HASHMAP_TOTAL_BUCKETS;
138  }

226  bool Get(const std::vector<uint8_t> vKey, SectorKey& cKey)
227  {
228  LOCK(KEY_MUTEX);
229 
230  /* Get the assigned bucket for the hashmap. */
231  uint32_t nBucket = GetBucket(vKey);
232 
233  /* Get the file binary position. */
234  uint32_t nFilePos = nBucket * HASHMAP_KEY_ALLOCATION;
235 
236  /* Reverse iterate the linked file list from hashmap to get most recent keys first. */
237  for(uint32_t i = hashmap[nBucket]; i >= 0; --i)
238  {
239  /* Find the file stream for LRU cache. */
240  std::fstream* pstream;
241  if(!fileCache->Get(i, pstream))
242  {
243  /* Set the new stream pointer. */
244  pstream = new std::fstream(strprintf("%s_hashmap.%05u", strBaseLocation.c_str(), i), std::ios::in | std::ios::out | std::ios::binary);
245 
246  /* If file not found add to LRU cache. */
247  fileCache->Put(i, pstream);
248  }
249 
250  /* Seek to the hashmap index in file. */
251  pstream->seekg (nFilePos, std::ios::beg);
252 
253  /* Read the bucket binary data from file stream */
254  std::vector<uint8_t> vBucket(HASHMAP_KEY_ALLOCATION, 0);
255  pstream->read((char*) &vBucket[0], vBucket.size());
256 
257  /* Check if this bucket has the key */
258  if(std::equal(vBucket.begin() + 11, vBucket.begin() + 11 + vKey.size(), vKey.begin()))
259  {
260  /* Deserialie key and return if found. */
261  DataStream ssKey(vBucket, SER_LLD, DATABASE_VERSION);
262  ssKey >> cKey;
263 
264  return true;
265  }
266  }
267 
268  return false;
269  }

281  bool Put(const SectorKey &cKey) const
282  {
283  LOCK(KEY_MUTEX);
284 
285  /* Get the assigned bucket for the hashmap. */
286  uint32_t nBucket = GetBucket(cKey.vKey);
287 
288  /* Get the file binary position. */
289  uint32_t nFilePos = nBucket * HASHMAP_KEY_ALLOCATION;
290 
291  /* Create a new disk hashmap object in linked list if it doesn't exist. */
292  std::string file = strprintf("%s_hashmap.%05u", strBaseLocation.c_str(), hashmap[nBucket]);
293  if(!filesystem::exists(file))
294  {
295  /* Blank vector to write empty space in new disk file. */
296  std::vector<uint8_t> vSpace(HASHMAP_TOTAL_BUCKETS * HASHMAP_KEY_ALLOCATION, 0);
297 
298  /* Write the blank data to the new file handle. */
299  std::fstream stream(file, std::ios::out | std::ios::binary | std::ios::trunc);
300  stream.write((char*)&vSpace[0], vSpace.size());
301  stream.close();
302 
303  /* Debug output for monitoring new disk maps. */
304  debug::log(0, FUNCTION, "Generated Disk Hash Map ", hashmap[nBucket], " of ", vSpace.size(), " bytes");
305  }
306 
307  /* Find the file stream for LRU cache. */
308  std::fstream* pstream;
309  if(!fileCache->Get(hashmap[nBucket], pstream))
310  {
311  /* Set the new stream pointer. */
312  pstream = new std::fstream(file, std::ios::in | std::ios::out | std::ios::binary);
313 
314  /* If not in cache, add to the LRU. */
315  fileCache->Put(hashmap[nBucket], pstream);
316  }
317 
318  /* Iterate the linked list value in the hashmap. */
319  ++hashmap[nBucket];
320 
321  /* Read the State and Size of Sector Header. */
322  DataStream ssKey(SER_LLD, DATABASE_VERSION);
323  ssKey << cKey;
324 
325  /* Serialize the key into the end of the vector. */
326  ssKey.write((char*)&cKey.vKey[0], cKey.vKey.size());
327 
328  /* Flush the key file to disk. */
329  pstream->seekp (nFilePos, std::ios::beg);
330  pstream->write((char*)&ssKey[0], ssKey.size());
331 
332  return true;
333  }
