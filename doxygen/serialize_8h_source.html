250 inline uint32_t GetSizeOfCompactSize(uint64_t nSize)
251 {
252  if (nSize < 253)
253  return sizeof(uint8_t);
254  else if (nSize <= std::numeric_limits<uint16_t>::max())
255  return sizeof(uint8_t) + sizeof(uint16_t);
256  else if (nSize <= std::numeric_limits<uint32_t>::max())
257  return sizeof(uint8_t) + sizeof(uint32_t);
258  else
259  return sizeof(uint8_t) + sizeof(uint64_t);
260 }

271 template<typename Stream>
272 void WriteCompactSize(Stream& os, uint64_t nSize)
273 {
274  if (nSize < 253)
275  {
276  uint8_t chSize = nSize;
277  WRITEDATA(os, chSize);
278  }
279  else if (nSize <= std::numeric_limits<uint16_t>::max())
280  {
281  uint8_t chSize = 253;
282  uint16_t xSize = nSize;
283  WRITEDATA(os, chSize);
284  WRITEDATA(os, xSize);
285  }
286  else if (nSize <= std::numeric_limits<uint32_t>::max())
287  {
288  uint8_t chSize = 254;
289  uint32_t xSize = nSize;
290  WRITEDATA(os, chSize);
291  WRITEDATA(os, xSize);
292  }
293  else
294  {
295  uint8_t chSize = 255;
296  uint64_t xSize = nSize;
297  WRITEDATA(os, chSize);
298  WRITEDATA(os, xSize);
299  }
300  return;

313 template<typename Stream>
314 uint64_t ReadCompactSize(Stream& is)
315 {
316  uint8_t chSize;
317  READDATA(is, chSize);
318  uint64_t nSizeRet = 0;
319  if (chSize < 253)
320  {
321  nSizeRet = chSize;
322  }
323  else if (chSize == 253)
324  {
325  uint16_t xSize;
326  READDATA(is, xSize);
327  nSizeRet = xSize;
328  }
329  else if (chSize == 254)
330  {
331  uint32_t xSize;
332  READDATA(is, xSize);
333  nSizeRet = xSize;
334  }
335  else
336  {
337  uint64_t xSize;
338  READDATA(is, xSize);
339  nSizeRet = xSize;
340  }
341  if (nSizeRet > (uint64_t)MAX_SIZE)
342  THROW_WITH_STACKTRACE(std::ios_base::failure("ReadCompactSize() : size too large"));
343  return nSizeRet;
344 }

607 template<typename Stream, typename T, typename A>
608 void Unserialize_impl(Stream& is, std::vector<T, A>& v, uint32_t nSerType, uint32_t nSerVersion, const std::true_type&)
609 {
610  // Limit size per read so bogus size value won't cause out of memory
611  v.clear();
612  uint32_t nSize = ReadCompactSize(is);
613  uint32_t i = 0;
614  while (i < nSize)
615  {
616  uint32_t blk = std::min(nSize - i, (uint32_t)(1 + 4999999 / sizeof(T)));
617  v.resize(i + blk);
618  is.read((char*)&v[i], blk * sizeof(T));
619  i += blk;
620  }
621 }

623 template<typename Stream, typename T, typename A>
624 void Unserialize_impl(Stream& is, std::vector<T, A>& v, uint32_t nSerType, uint32_t nSerVersion, const std::false_type&)
625 {
626  v.clear();
627  uint32_t nSize = ReadCompactSize(is);
628  uint32_t i = 0;
629  uint32_t nMid = 0;
630  while (nMid < nSize)
631  {
632  nMid += 5000000 / sizeof(T);
633  if (nMid > nSize)
634  nMid = nSize;
635  v.resize(nMid);
636  for (; i < nMid; i++)
637  Unserialize(is, v[i], nSerType, nSerVersion);
638  }
