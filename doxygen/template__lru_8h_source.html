213  uint32_t Bucket(const KeyType &Key) const
214  {
215  /* Get the bytes from the type object. */
216  std::vector<uint8_t> vKey;
217  vKey.insert(vKey.end(), (uint8_t*)&Key, (uint8_t*)&Key + sizeof(Key));
218  uint64_t s = vKey.size();
219 
220  /* Find the bucket through creating an uint64_t from available bytes. */
221  uint64_t nBucket = 0;
222  for(uint64_t i = 0; i < s && i < 8; ++i)
223  nBucket += vKey[i] << (8 * i);
224 
225  /* Round robin to find the bucket. */
226  return nBucket % MAX_CACHE_BUCKETS;
227  }

292  void MoveToFront(TemplateNode<KeyType, DataType> *pthis)
293  {
294  /* Don't move to front if already in the front. */
295  if(pthis == pfirst)
296  return;
297 
298  /* Move last pointer if moving from back. */
299  if(pthis == plast)
300  {
301  if(plast->pprev)
302  plast = plast->pprev;
303 
304  plast->pnext = nullptr;
305  }
306  else

310  pthis->pprev = nullptr;

313  pthis->pnext = pfirst;
314
