205  debug::log(0, FUNCTION, "repairing index for ", hashTransaction.ToString().substr(0, 20));
206 
207  /* Get the best block state to start from. */
208  TAO::Ledger::BlockState state = TAO::Ledger::ChainState::stateBest.Prev();
209 
210  /* Loop until it is found. */
211  while(!config::fShutdown && !state.IsNull())
212  {
213  /* Give debug output of status. */
214  if(state.nHeight % 100000 == 0)
215  debug::log(0, FUNCTION, "repairing index..... ", state.nHeight);
216 
217  /* Check for the transaction. */
218  for(const auto& tx : state.vtx)
219  {
220  /* If the transaction is found, write the index. */

364  bool WriteProof(const uint256_t& hashProof, const uint512_t& hashTransaction, uint8_t nFlags = TAO::Register::FLAGS::WRITE)
365  {
366  /* Memory mode for pre-database commits. */
367  if(nFlags & TAO::Register::FLAGS::MEMPOOL)
368  {
369  LOCK(MEMORY_MUTEX);
370 
371  /* Write the new proof state. */
372  mapProofs[std::make_pair(hashProof, hashTransaction)] = 0;
373  return true;
374  }
375  else
376  {
377  LOCK(MEMORY_MUTEX);
378 
379  /* Erase memory proof if they exist. */
380  if(mapProofs.count(std::make_pair(hashProof, hashTransaction)))
381  mapProofs.erase(std::make_pair(hashProof, hashTransaction));
382  }
383 
384  return Write(std::make_pair(hashProof, hashTransaction));
