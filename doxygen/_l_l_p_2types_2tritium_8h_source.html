134  void ReadPacket() final
135  {
136  if(!INCOMING.Complete())
137  {
139  if(INCOMING.IsNull() && Available() >= 10)
140  {
141  std::vector<uint8_t> BYTES(10, 0);
142  if(Read(BYTES, 10) == 10)
143  {
144  DataStream ssHeader(BYTES, SER_NETWORK, MIN_PROTO_VERSION);
145  ssHeader >> INCOMING;
146 
147  Event(EVENT_HEADER);
148  }
149  }
150 
152  uint32_t nAvailable = Available();
153  if(nAvailable > 0 && !INCOMING.IsNull() && INCOMING.DATA.size() < INCOMING.LENGTH)
154  {
155  /* Create the packet data object. */
156  std::vector<uint8_t> DATA( std::min( nAvailable, (uint32_t)(INCOMING.LENGTH - INCOMING.DATA.size())), 0);
157 
158  /* Read up to 512 bytes of data. */
159  if(Read(DATA, DATA.size()) == DATA.size())
160  {
161  INCOMING.DATA.insert(INCOMING.DATA.end(), DATA.begin(), DATA.end());
162  Event(EVENT_PACKET, DATA.size());
163  }
164  }
165  }
