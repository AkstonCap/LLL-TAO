201  void Thread()
202  {
203  /* The mutex for the condition. */
204  std::mutex CONDITION_MUTEX;
205 
206  /* The main connection handler loop. */
207  while(!fDestruct.load())
208  {
209  /* Keep thread from consuming too many resources. */
210  runtime::sleep(1);
211 
212  /* Keep data threads waiting for work. */
213  std::unique_lock<std::mutex> CONDITION_LOCK(CONDITION_MUTEX);
214  CONDITION.wait(CONDITION_LOCK, [this]{ return fDestruct.load() || nConnections.load() > 0; });
215 
216  /* Check for close. */
217  if(fDestruct.load())
218  return;
219 
220  { LOCK(MUTEX);
221 
222  /* Poll the sockets. */
223  int nPoll = poll((pollfd*)CONNECTIONS[0], CONNECTIONS.size(), 100);
224  if(nPoll < 0)
225  continue;
226  }
227 
228  /* Check all connections for data and packets. */
229  uint32_t nSize = static_cast<uint32_t>(CONNECTIONS.size());
230  for(uint32_t nIndex = 0; nIndex < nSize; ++nIndex)
231  {
232  try
233  {
234  /* Skip over Inactive Connections. */
235  if(CONNECTIONS[nIndex]->IsNull() || !CONNECTIONS[nIndex]->Connected())
236  continue;
237 
238  /* Remove Connection if it has Timed out or had any Errors. */
239  if(CONNECTIONS[nIndex]->Errors())
240  {
241  disconnect_remove_event(nIndex, DISCONNECT_ERRORS);
242  continue;
243  }
244 
245  /* Remove Connection if it has Timed out or had any Errors. */
246  if(CONNECTIONS[nIndex]->Timeout(TIMEOUT))
247  {
248  disconnect_remove_event(nIndex, DISCONNECT_TIMEOUT);
249  continue;
250  }
251 
252  /* Handle any DDOS Filters. */
253  if(fDDOS)
254  {
255  /* Ban a node if it has too many Requests per Second. **/
256  if(CONNECTIONS[nIndex]->DDOS->rSCORE.Score() > DDOS_rSCORE ||
257  CONNECTIONS[nIndex]->DDOS->cSCORE.Score() > DDOS_cSCORE)
258  CONNECTIONS[nIndex]->DDOS->Ban();
259 
260  /* Remove a connection if it was banned by DDOS Protection. */
261  if(CONNECTIONS[nIndex]->DDOS->Banned())
262  {
263  disconnect_remove_event(nIndex, DISCONNECT_DDOS);
264  continue;
265  }
266  }
267 
268  /* Generic event for Connection. */
269  CONNECTIONS[nIndex]->Event(EVENT_GENERIC);
270 
271  /* Flush the write buffer. */
272  CONNECTIONS[nIndex]->Flush();
273 
274  /* Work on Reading a Packet. **/
275  CONNECTIONS[nIndex]->ReadPacket();
276 
277  /* If a Packet was received successfully, increment request count [and DDOS count if enabled]. */
278  if(CONNECTIONS[nIndex]->PacketComplete())
279  {
280  /* Debug dump of message type. */
281  debug::log(4, FUNCTION, "Recieved Message (", CONNECTIONS[nIndex]->INCOMING.GetBytes().size(), " bytes)");
282 
283  /* Debug dump of packet data. */
284  if(config::GetArg("-verbose", 0) >= 5)
285  PrintHex(CONNECTIONS[nIndex]->INCOMING.GetBytes());
286 
287  /* Handle Meters and DDOS. */
288  if(fMETER)
289  ++REQUESTS;
290  if(fDDOS)
291  CONNECTIONS[nIndex]->DDOS->rSCORE += 1;
292 
293  /* Packet Process return value of False will flag Data Thread to Disconnect. */
294  if(!CONNECTIONS[nIndex]->ProcessPacket())
295  {
296  disconnect_remove_event(nIndex, DISCONNECT_FORCE);
297  continue;
298  }
299 
300  CONNECTIONS[nIndex]->ResetPacket();
301  }
302  }
303  catch(std::exception& e)
304  {
305  debug::error(FUNCTION, "data connection: ", e.what());
306  disconnect_remove_event(nIndex, DISCONNECT_ERRORS);
307  }
308  }
