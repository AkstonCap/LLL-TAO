217  void CompressKey(std::vector<uint8_t>& vData, uint16_t nSize = 32)
218  {
219  /* Loop until key is of desired size. */
220  while(vData.size() > nSize)
221  {
222  /* Loop half of the key to XOR elements. */
223  for(uint64_t i = 0; i < vData.size() / 2; ++i)
224  if(i * 2 < vData.size())
225  vData[i] ^= vData[i * 2];
226 
227  /* Resize the container to half its size. */
228  vData.resize(vData.size() / 2);
229  }
230  }

255  uint32_t GetBucket(const std::vector<uint8_t>& vKey)
256  {
257  /* Get an MD5 digest. */
258  uint8_t digest[MD5_DIGEST_LENGTH];
259  MD5((uint8_t *)&vKey[0], vKey.size(), (uint8_t *)&digest);
260 
261  /* Copy bytes into the bucket. */
262  uint64_t nBucket;
263  std::copy((uint8_t *)&digest[0], (uint8_t *)&digest[0] + 8, (uint8_t *)&nBucket);
264 
265  return nBucket % HASHMAP_TOTAL_BUCKETS;
266  }

274  void Initialize()
275  {
276  /* Create directories if they don't exist yet. */
277  if(!filesystem::exists(strBaseLocation) && filesystem::create_directories(strBaseLocation))
278  debug::log(0, FUNCTION, "Generated Path ", strBaseLocation);

280  /* Build the hashmap indexes. */
281  std::string index = debug::strprintf("%s_hashmap.index", strBaseLocation.c_str());

283  {
284  /* Generate empty space for new file. */
285  std::vector<uint8_t> vSpace(HASHMAP_TOTAL_BUCKETS * 4, 0);
286 
287  /* Write the new disk index .*/
288  std::fstream stream(index, std::ios::out | std::ios::binary | std::ios::trunc);
289  stream.write((char*)&vSpace[0], vSpace.size());
290  stream.close();
291 
292  /* Debug output showing generation of disk index. */
293  debug::log(0, FUNCTION, "Generated Disk Index of ", vSpace.size(), " bytes");
294  }
295 
296  /* Read the hashmap indexes. */
297  else
298  {
299  /* Build a vector to read the disk index. */
300  std::vector<uint8_t> vIndex(HASHMAP_TOTAL_BUCKETS * 4, 0);
301 
302  /* Read the disk index bytes. */
303  std::fstream stream(index, std::ios::in | std::ios::binary);
304  stream.read((char*)&vIndex[0], vIndex.size());
305  stream.close();
306 
307  /* Deserialize the values into memory index. */
308  uint32_t nTotalKeys = 0;
309  for(uint32_t nBucket = 0; nBucket < HASHMAP_TOTAL_BUCKETS; ++nBucket)
310  {
311  std::copy((uint8_t *)&vIndex[nBucket * 4], (uint8_t *)&vIndex[nBucket * 4] + 4, (uint8_t *)&hashmap[nBucket]);
312 
313  nTotalKeys += hashmap[nBucket];
314  }
315 
316  /* Debug output showing loading of disk index. */
317  debug::log(0, FUNCTION, "Loaded Disk Index of ", vIndex.size(), " bytes and ", nTotalKeys, " keys");
318  }
319 
320  /* Build the first hashmap index file if it doesn't exist. */
321  std::string file = debug::strprintf("%s_hashmap.%05u", strBaseLocation.c_str(), 0u).c_str();
322  if(!filesystem::exists(file))
323  {
324  /* Build a vector with empty bytes to flush to disk. */
325  std::vector<uint8_t> vSpace(HASHMAP_TOTAL_BUCKETS * HASHMAP_KEY_ALLOCATION, 0);

328  std::fstream stream(file, std::ios::out | std::ios::binary | std::ios::trunc);
329  stream.write((char*)&vSpace[0], vSpace.size());

332  /* Debug output showing generating of the hashmap file. */
333  debug::log(0, FUNCTION, "Generated Disk Hash Map 0 of ", vSpace.size(), " bytes");
334  }
335 
336  /* Load the stream object into the stream LRU cache. */
337  fileCache->Put(0, new std::fstream(file, std::ios::in | std::ios::out | std::ios::binary));

358  {
359  /* Get the assigned bucket for the hashmap. */

362  /* Get the file binary position. */

370  CompressKey(vKeyCompressed, HASHMAP_MAX_KEY_SIZE);
371 
372  /* Reverse iterate the linked file list from hashmap to get most recent keys first. */
373  std::vector<uint8_t> vBucket(HASHMAP_KEY_ALLOCATION, 0);
374  for(int i = hashmap[nBucket] - 1; i >= 0; --i)
375  {
376  { LOCK(KEY_MUTEX);
377 
378  /* Find the file stream for LRU cache. */
379  std::fstream *pstream;
380  if(!fileCache->Get(i, pstream))

382  /* Set the new stream pointer. */
383  std::string filename = debug::strprintf("%s_hashmap.%05u", strBaseLocation.c_str(), i);
384 
385  pstream = new std::fstream(filename, std::ios::in | std::ios::out | std::ios::binary);
386  if(!pstream->is_open())
387  {
388  delete pstream;
389  return debug::error(FUNCTION, "couldn't create hashmap object at: ",
390  filename, " (", strerror(errno), ")");
391  }
392 
393  /* If file not found add to LRU cache. */
394  fileCache->Put(i, pstream);

396 
397  /* Seek to the hashmap index in file. */
398  pstream->seekg(nFilePos, std::ios::beg);
399

418  " | Sector Size: ", cKey.nSectorSize,
