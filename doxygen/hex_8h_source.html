150 template<typename T>
151 std::string HexStr(const T itbegin, const T itend, bool fSpaces=false)
152 {
153  std::vector<char> rv;
154  static char hexmap[16] = { '0', '1', '2', '3', '4', '5', '6', '7',
155  '8', '9', 'a', 'b', 'c', 'd', 'e', 'f' };
156  rv.reserve((itend-itbegin)*3);
157 
158  int nTotal = 0;
159  for(T it = itbegin; it < itend; ++it, ++nTotal)
160  {
161  uint8_t val = (uint8_t)(*it);
162  if(fSpaces && it != itbegin && nTotal % 32 == 0)
163  rv.push_back('\n');
164  else if(fSpaces && it != itbegin && it != itend && nTotal % 4 == 0)
165  rv.push_back(' ');
166  rv.push_back(hexmap[val>>4]);
167  rv.push_back(hexmap[val&15]);
168  }
169 
170  return std::string(rv.begin(), rv.end());
171 }
