164 /* These defines are used to declare buffers in a way that allows
165  faster operations on longer variables to be used. In all these
166  defines 'size' must be a power of 2 and >= 8
167 
168  dec_unit_type(size,x) declares a variable 'x' of length
169  'size' bits
170 
171  dec_bufr_type(size,bsize,x) declares a buffer 'x' of length 'bsize'
172  bytes defined as an array of variables
173  each of 'size' bits (bsize must be a
174  multiple of size / 8)
175 
176  ptr_cast(x,size) casts a pointer to a pointer to a
177  varaiable of length 'size' bits
178 */
179 
180 #define ui_type(size) uint_##size##t
181 #define dec_unit_type(size,x) typedef ui_type(size) x
182 #define dec_bufr_type(size,bsize,x) typedef ui_type(size) x[bsize / (size >> 3)]
183 #define ptr_cast(x,size) ((ui_type(size)*)(x))
