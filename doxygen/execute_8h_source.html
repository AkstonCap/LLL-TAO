49  inline bool Execute(TAO::Ledger::Transaction& tx, uint8_t nFlags)
50  {
51  /* Get the caller. */
52  uint256_t hashOwner = tx.hashGenesis;
53 
54  /* Start the stream at the beginning. */
55  tx.ssOperation.seek(0, STREAM::BEGIN);
56 
57  /* Start the register stream at the beginning. */
58  tx.ssRegister.seek(0, STREAM::BEGIN);
59 
60 
61  /* Make sure no exceptions are thrown. */
62  try
63  {
64 
65  /* Loop through the operations tx.ssOperation. */
66  while(!tx.ssOperation.end())
67  {
68  uint8_t OPERATION;
69  tx.ssOperation >> OPERATION;
70 
71  /* Check the current opcode. */
72  switch(OPERATION)
73  {
74 
75  /* Record a new state to the register. */
76  case OP::WRITE:
77  {
78  /* Get the Address of the Register. */
79  uint256_t hashAddress;
80  tx.ssOperation >> hashAddress;
81 
82  /* Deserialize the register from tx.ssOperation. */
83  std::vector<uint8_t> vchData;
84  tx.ssOperation >> vchData;
85 
86  /* Execute the operation method. */
87  if(!Write(hashAddress, vchData, hashOwner, nFlags, tx))
88  return false;
89 
90  break;
91  }
92 
93 
94  /* Append a new state to the register. */
95  case OP::APPEND:
96  {
97  /* Get the Address of the Register. */
98  uint256_t hashAddress;

100 
101  /* Deserialize the register from tx.ssOperation. */

104

106  if(!Append(hashAddress, vchData, hashOwner, nFlags, tx))

112

114  case OP::REGISTER:

117  uint256_t hashAddress;
